#include <Adafruit_Sensor.h>
#include <DHT.h>

// กำหนดพินที่ใช้เชื่อมต่อกับ DHT22
#define DHTPIN 2     // เปลี่ยนเป็นพินที่ใช้
#define DHTTYPE DHT22  // ใช้ DHT22 หรือเปลี่ยนเป็น DHT11 ถ้าใช้ DHT11

DHT dht(DHTPIN, DHTTYPE); // สร้างอ็อบเจ็กต์ DHT

void setup() {
  Serial.begin(115200);  // เริ่มต้น Serial Monitor
  dht.begin();  // เริ่มต้นเซ็นเซอร์ DHT
}

void loop() {
  // อ่านค่าอุณหภูมิและความชื้น
  float humidity = dht.readHumidity();
  float temperature = dht.readTemperature(); // ค่าเป็นองศาเซลเซียส

  // ตรวจสอบว่าการอ่านค่ามีปัญหาหรือไม่
  if (isnan(humidity) || isnan(temperature)) {
    Serial.println("No Detect Sensor");
    return;
  }

  // แสดงผลใน Serial Monitor
  Serial.print("Temperature:");
  Serial.print(temperature);
  Serial.println("C");
  Serial.print("Humidity:");
  Serial.print(humidity);
  Serial.println("%");

  delay(5000);  // อ่านค่าทุกๆ 2 วินาที
}
